#### 이차원배열의 행의 합을 구하여 일차원배열에 저장하는 함수의 정의   
<br/>
<br/>
### - code
```c++
#include <iostream>
#include <iomanip>

using namespace std;

const int ROW=3, COL=3;

// 이차원배열의 행의 합을 구하여 일차원배열에 저장하는 함수의 정의
// 결과는 모두 동일

// 이차원 배열의 행을 포인터 배열로 전달하여 행을 직접 참조
void row_sum_1(const int (*array)[COL], int *ptr) {
    for (int i=0; i<ROW; i++)
        for (int j=1; j<COL; j++)
            *(ptr+i) += *(*(array+i)+j);
}

// 단일 행을 1차원 배열의 포인터로 전달
void row_sum_2(const int *rowarr, int *ptr) {
    for (int j=1; j<COL; j++)
        *ptr += *(rowarr+j);
}

// 이차원 배열을 일차원 포인터로 전달하여 전체 배열을 일차원으로 처리
void row_sum_3(const int *pt, int *ptr) {
    for (int i=0; i<ROW; i++)
        for (int j=1; j<COL; j++)
            *(ptr+i) += *(pt + (COL*i+j));
}

int main() 
{
    int a[ROW][COL] = {{1, 10, 20}, {2, 25, 35}, {3, 30, 40}};
    int rsum[ROW] = {0};

    row_sum_1(a, rsum);

    cout << " row_num |   Matrix   | a[i][1]+a[i][2] " << "\n";
    cout << "-------------------------------------" << "\n";

    for (int i=0; i<ROW; i++) {
        cout << "    " << i << "    | ";
        for (int j=0; j<COL; j++)
            cout << setw(3) << a[i][j];
        cout << "  |      " << rsum[i] << "\n";
    }

    return 0;
}
```
